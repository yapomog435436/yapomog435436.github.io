Просто скопируйте необходимое и вставьте в свой блог или сайт.

Для блога: 

<script language="JavaScript" src="http://ваш_сайт.ru/wp-content/themes/ваша_тема/audio/audio-player.js"></script>
 <object type="application/x-shockwave-flash" data="http://ваш_сайт.ru/wp-content/themes/ваша_тема/audio/player.swf" id="audioplayer1" height="24" width="290">
 <param name="movie" value="http://ваш_сайт.ru/wp-content/themes/ваша_тема/audio/player.swf">
 <param name="FlashVars" value="playerID=1&amp;soundFile=http://ваш_сайт.ru/wp-content/themes/ваша_тема/audio/ваш_файл.mp3">
 <param name="quality" value="high">
 <param name="menu" value="false">
 <param name="wmode" value="transparent">
 </object>
